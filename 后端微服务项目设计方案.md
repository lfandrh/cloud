模块职责

| 模块                | 职责                                                         |
| ------------------- | ------------------------------------------------------------ |
| **common**          | Token 解析、当前用户信息工具、统一返回对象、异常、常量、工具类 |
| **gateway-service** | 请求入口、Token 校验、用户信息透传、路由转发、限流/熔断      |
| **auth-service**    | 登录/登出、Token 发放、权限/角色管理、Token 校验接口         |
| **user-service**    | 用户业务逻辑、用户信息 CRUD、依赖 common 获取当前用户信息    |

关键，所有的Token解析和校验都在common模块中

### 完整SQL

```sql
-- ===============================================
-- 数据库：cloud
-- ===============================================
CREATE DATABASE IF NOT EXISTS cloud DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_general_ci;
USE cloud;

-- ===============================================
-- 1️⃣ 用户表 sys_user
-- ===============================================
CREATE TABLE IF NOT EXISTS `sys_user` (
    `id` BIGINT AUTO_INCREMENT PRIMARY KEY,
    `user_name` VARCHAR(50) NOT NULL UNIQUE COMMENT '登录用户名',
    `nick_name` VARCHAR(50) COMMENT '昵称',
    `gender` TINYINT DEFAULT 1 COMMENT '性别：1男, 2女',
    `phone` VARCHAR(20) UNIQUE COMMENT '手机号',
    `email` VARCHAR(100) UNIQUE COMMENT '邮箱',
    `status` TINYINT DEFAULT 1 COMMENT '启用状态：1启用, 2禁用',
    `password` VARCHAR(255) NOT NULL COMMENT '登录密码（加密存储）',
    `create_by` VARCHAR(50),
    `create_time` DATETIME DEFAULT CURRENT_TIMESTAMP,
    `update_by` VARCHAR(50),
    `update_time` DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户表';

-- ===============================================
-- 2️⃣ 角色表 sys_role
-- ===============================================
CREATE TABLE IF NOT EXISTS `sys_role` (
    `id` BIGINT AUTO_INCREMENT PRIMARY KEY,
    `role_name` VARCHAR(50) NOT NULL UNIQUE COMMENT '角色名称',
    `role_code` VARCHAR(50) NOT NULL UNIQUE COMMENT '角色编码',
    `role_desc` VARCHAR(255) COMMENT '角色描述',
    `status` TINYINT DEFAULT 1 COMMENT '启用状态：1启用, 2禁用',
    `create_by` VARCHAR(50),
    `create_time` DATETIME DEFAULT CURRENT_TIMESTAMP,
    `update_by` VARCHAR(50),
    `update_time` DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='角色表';

-- ===============================================
-- 3️⃣ 用户-角色关联表 sys_user_role
-- ===============================================
CREATE TABLE IF NOT EXISTS `sys_user_role` (
    `id` BIGINT AUTO_INCREMENT PRIMARY KEY,
    `user_id` BIGINT NOT NULL COMMENT '用户ID',
    `role_id` BIGINT NOT NULL COMMENT '角色ID',
    UNIQUE KEY `uniq_user_role` (`user_id`,`role_id`),
    CONSTRAINT `fk_sysuserrole_user` FOREIGN KEY (`user_id`) REFERENCES `sys_user` (`id`) ON DELETE CASCADE,
    CONSTRAINT `fk_sysuserrole_role` FOREIGN KEY (`role_id`) REFERENCES `sys_role` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户角色关联表';

-- ===============================================
-- 4️⃣ 菜单表 sys_menu
-- ===============================================
CREATE TABLE IF NOT EXISTS `sys_menu` (
    `id` BIGINT AUTO_INCREMENT PRIMARY KEY,
    `parent_id` BIGINT DEFAULT 0 COMMENT '父菜单ID，0表示顶级菜单',
    `menu_type` TINYINT NOT NULL COMMENT '菜单类型：1目录, 2菜单',
    `menu_name` VARCHAR(50) NOT NULL COMMENT '菜单名称',
    `route_name` VARCHAR(50) COMMENT '路由名称',
    `route_path` VARCHAR(255) COMMENT '路由路径',
    `component` VARCHAR(255) COMMENT '组件路径',
    `icon` VARCHAR(50) COMMENT '图标',
    `icon_type` TINYINT DEFAULT 1 COMMENT '图标类型：1=iconify, 2=local',
    `order_num` INT DEFAULT 0 COMMENT '排序',
    `status` TINYINT DEFAULT 1 COMMENT '启用状态：1启用, 2禁用',
    `create_by` VARCHAR(50),
    `create_time` DATETIME DEFAULT CURRENT_TIMESTAMP,
    `update_by` VARCHAR(50),
    `update_time` DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='菜单表';

-- ===============================================
-- 5️⃣ 菜单按钮表 sys_menu_button
-- ===============================================
CREATE TABLE IF NOT EXISTS `sys_menu_button` (
    `id` BIGINT AUTO_INCREMENT PRIMARY KEY,
    `menu_id` BIGINT NOT NULL COMMENT '所属菜单ID',
    `button_name` VARCHAR(50) NOT NULL COMMENT '按钮名称',
    `button_code` VARCHAR(50) NOT NULL COMMENT '按钮编码',
    `status` TINYINT DEFAULT 1 COMMENT '启用状态：1启用, 2禁用',
    CONSTRAINT `fk_sysmenubutton_menu` FOREIGN KEY (`menu_id`) REFERENCES `sys_menu` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='菜单按钮表';

-- ===============================================
-- 6️⃣ 角色-菜单关联表 sys_role_menu
-- ===============================================
CREATE TABLE IF NOT EXISTS `sys_role_menu` (
    `id` BIGINT AUTO_INCREMENT PRIMARY KEY,
    `role_id` BIGINT NOT NULL COMMENT '角色ID',
    `menu_id` BIGINT NOT NULL COMMENT '菜单ID',
    UNIQUE KEY `uniq_role_menu` (`role_id`,`menu_id`),
    CONSTRAINT `fk_sysrolemenu_role` FOREIGN KEY (`role_id`) REFERENCES `sys_role` (`id`) ON DELETE CASCADE,
    CONSTRAINT `fk_sysrolemenu_menu` FOREIGN KEY (`menu_id`) REFERENCES `sys_menu` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='角色菜单关联表';

-- ===============================================
-- 7️⃣ 动态路由表 sys_route
-- ===============================================
CREATE TABLE IF NOT EXISTS `sys_route` (
    `id` BIGINT AUTO_INCREMENT PRIMARY KEY,
    `parent_id` BIGINT DEFAULT 0 COMMENT '父路由ID，0表示顶级路由',
    `route_name` VARCHAR(50) NOT NULL COMMENT '路由名称',
    `path` VARCHAR(255) NOT NULL COMMENT '路由路径',
    `component` VARCHAR(255) COMMENT '组件路径',
    `icon` VARCHAR(50) COMMENT '图标',
    `menu_type` TINYINT DEFAULT 2 COMMENT '路由类型：1目录, 2页面',
    `order_num` INT DEFAULT 0 COMMENT '排序'
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='动态路由表';
```
